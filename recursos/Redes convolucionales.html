üíª 6. CNN en Acci√≥n: Clasificador de Im√°genes con Python

Vamos a construir una CNN para clasificar im√°genes del dataset CIFAR-10, que contiene 10 clases (aviones, coches, p√°jaros, etc.).




    üêç CNN para clasificar CIFAR-10 con TensorFlow/Keras
    


import tensorflow as tf
from tensorflow.keras import datasets, layers, models

# --- 1. CARGAR Y PREPARAR DATOS ---
(train_images, train_labels), (test_images, test_labels) = datasets.cifar10.load_data()

# Normalizar los valores de los p√≠xeles para que est√©n entre 0 y 1
train_images, test_images = train_images / 255.0, test_images / 255.0

class_names = ['airplane', 'automobile', 'bird', 'cat', 'deer',
               'dog', 'frog', 'horse', 'ship', 'truck']

# --- 2. CONSTRUIR LA ARQUITECTURA CNN ---
model = models.Sequential()
# Bloque 1
model.add(layers.Conv2D(32, (3, 3), activation='relu', input_shape=(32, 32, 3)))
model.add(layers.MaxPooling2D((2, 2)))
# Bloque 2
model.add(layers.Conv2D(64, (3, 3), activation='relu'))
model.add(layers.MaxPooling2D((2, 2)))
# Bloque 3
model.add(layers.Conv2D(64, (3, 3), activation='relu'))

# Aplanar y a√±adir capas densas para la clasificaci√≥n
model.add(layers.Flatten())
model.add(layers.Dense(64, activation='relu'))
model.add(layers.Dense(10)) # 10 clases de salida

model.summary()

# --- 3. COMPILAR Y ENTRENAR ---
model.compile(optimizer='adam',
              loss=tf.keras.losses.SparseCategoricalCrossentropy(from_logits=True),
              metrics=['accuracy'])

# Para una demo, solo entrenamos 5 √©pocas. En un caso real, ser√≠an m√°s.
history = model.fit(train_images, train_labels, epochs=5, 
                    validation_data=(test_images, test_labels))

# --- 4. EVALUAR ---
test_loss, test_acc = model.evaluate(test_images,  test_labels, verbose=2)
print(f"\nPrecisi√≥n en el conjunto de prueba: {test_acc:.2f}")
